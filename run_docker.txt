# Docker 開啟 OR 關閉

# 1. 停止並刪除舊容器
docker stop genshin_god_field && docker rm genshin_god_field

# 2. 從 GitHub 抓取最新程式碼
git pull

# 3. 重新建置映像檔 (如果代碼有變)
docker build -t genshin_god_field .

# 4. 重新啟動容器 (記得帶上掛載與環境變數)
docker run -d --name genshin_god_field --restart always genshin_god_field

# 查看執行紀錄，按下 ESC 退出
docker logs -f genshin_god_field
